# New API 实现粘性会话和客户端限制的技术方案

## 📋 执行摘要

本文档评估在 New API 中实现以下两个 CRS 特性的可行性：
1. **粘性会话（Sticky Session）**：同一会话/对话使用同一渠道
2. **客户端限制（Client Restriction）**：基于 User-Agent 限制 API Key 访问

**结论**：✅ **两个功能都可以实现**，且实现难度适中，不需要大规模重构。

---

## 🎯 功能 1: 粘性会话（Sticky Session）

### 现状分析

**当前 New API 的渠道选择逻辑** (middleware/distributor.go:100):

```go
// 每次请求都重新随机选择渠道
channel, selectGroup, err = service.CacheGetRandomSatisfiedChannel(c, usingGroup, modelRequest.Model, 0)
```

**问题**：
- 每次请求独立选择渠道
- 同一对话的多个请求可能使用不同渠道
- 可能导致上下文丢失或行为不一致

### 实现方案

#### 方案 A：基于 User 标识的会话绑定 ⭐⭐⭐⭐⭐ (推荐)

**设计原理**：
- 利用 OpenAI API 的 `user` 字段作为会话标识
- 为每个 user 绑定一个渠道，并维持一段时间
- 存储在 Redis 或内存缓存中

**数据结构**：

```go
// 会话到渠道的映射
type SessionChannelBinding struct {
    UserId      string    // user 字段值或生成的会话 ID
    ChannelId   int       // 绑定的渠道 ID
    Model       string    // 模型名称
    Group       string    // 渠道分组
    CreatedAt   time.Time // 创建时间
    LastUsedAt  time.Time // 最后使用时间
    ExpiresAt   time.Time // 过期时间
}

// Redis Key: session:channel:{userId}:{model}
// Value: channelId
// TTL: 1-24 小时（可配置）
```

**实现步骤**：

**1. 修改 Token 模型，添加粘性会话配置**


**2. 创建会话管理服务**

```go
// service/session.go
package service

import (
    "context"
    "fmt"
    "time"
    "github.com/QuantumNous/new-api/common"
    "github.com/QuantumNous/new-api/model"
)

type SessionManager struct{}

// GetSessionKey 生成会话的 Redis Key
func (sm *SessionManager) GetSessionKey(userId, modelName, group string) string {
    return fmt.Sprintf("session:channel:%s:%s:%s", userId, modelName, group)
}

// GetBoundChannel 获取会话绑定的渠道
func (sm *SessionManager) GetBoundChannel(userId, modelName, group string) (int, bool) {
    if !common.RedisEnabled {
        // 如果 Redis 未启用，使用内存缓存
        return sm.getFromMemoryCache(userId, modelName, group)
    }

    key := sm.GetSessionKey(userId, modelName, group)
    val, err := common.RedisGet(key)
    if err != nil {
        return 0, false
    }

    channelId, err := strconv.Atoi(val)
    if err != nil {
        return 0, false
    }

    return channelId, true
}

// BindChannel 绑定会话到渠道
func (sm *SessionManager) BindChannel(userId, modelName, group string, channelId int, ttl time.Duration) error {
    if !common.RedisEnabled {
        return sm.saveToMemoryCache(userId, modelName, group, channelId, ttl)
    }

    key := sm.GetSessionKey(userId, modelName, group)
    val := fmt.Sprintf("%d", channelId)
    return common.RedisSet(key, val, ttl)
}

// UpdateLastUsed 更新会话最后使用时间（延长 TTL）
func (sm *SessionManager) UpdateLastUsed(userId, modelName, group string, channelId int, ttl time.Duration) error {
    // 重新设置过期时间
    return sm.BindChannel(userId, modelName, group, channelId, ttl)
}

// UnbindChannel 解绑会话（渠道失败时调用）
func (sm *SessionManager) UnbindChannel(userId, modelName, group string) error {
    if !common.RedisEnabled {
        return sm.deleteFromMemoryCache(userId, modelName, group)
    }

    key := sm.GetSessionKey(userId, modelName, group)
    return common.RedisDel(key)
}

// 内存缓存实现（Redis 未启用时的降级方案）
var memorySessionCache = make(map[string]*SessionCacheItem)
var memorySessionMutex sync.RWMutex

type SessionCacheItem struct {
    ChannelId int
    ExpiresAt time.Time
}

func (sm *SessionManager) getFromMemoryCache(userId, modelName, group string) (int, bool) {
    memorySessionMutex.RLock()
    defer memorySessionMutex.RUnlock()

    key := sm.GetSessionKey(userId, modelName, group)
    item, exists := memorySessionCache[key]
    if !exists {
        return 0, false
    }

    if time.Now().After(item.ExpiresAt) {
        return 0, false
    }

    return item.ChannelId, true
}

func (sm *SessionManager) saveToMemoryCache(userId, modelName, group string, channelId int, ttl time.Duration) error {
    memorySessionMutex.Lock()
    defer memorySessionMutex.Unlock()

    key := sm.GetSessionKey(userId, modelName, group)
    memorySessionCache[key] = &SessionCacheItem{
        ChannelId: channelId,
        ExpiresAt: time.Now().Add(ttl),
    }
    return nil
}

func (sm *SessionManager) deleteFromMemoryCache(userId, modelName, group string) error {
    memorySessionMutex.Lock()
    defer memorySessionMutex.Unlock()

    key := sm.GetSessionKey(userId, modelName, group)
    delete(memorySessionCache, key)
    return nil
}

// 定期清理过期的内存缓存
func CleanupExpiredSessions() {
    ticker := time.NewTicker(5 * time.Minute)
    go func() {
        for range ticker.C {
            memorySessionMutex.Lock()
            now := time.Now()
            for key, item := range memorySessionCache {
                if now.After(item.ExpiresAt) {
                    delete(memorySessionCache, key)
                }
            }
            memorySessionMutex.Unlock()
        }
    }()
}
```

**3. 修改渠道选择逻辑**

```go
// middleware/distributor.go
func Distribute() func(c *gin.Context) {
    return func(c *gin.Context) {
        // ... 现有代码

        if shouldSelectChannel {
            // 检查是否启用粘性会话
            stickySessionEnabled := common.GetContextKeyBool(c, constant.ContextKeyStickySession)

            if stickySessionEnabled {
                // 尝试从会话获取绑定的渠道
                userId := getSessionUserId(c, modelRequest)
                sessionManager := &service.SessionManager{}

                if channelId, exists := sessionManager.GetBoundChannel(userId, modelRequest.Model, usingGroup); exists {
                    // 使用绑定的渠道
                    channel, err = model.GetChannelById(channelId, true)
                    if err == nil && channel.Status == common.ChannelStatusEnabled {
                        common.SetContextKey(c, constant.ContextKeyStickySessionUsed, true)
                        // 更新最后使用时间
                        ttl := common.GetContextKeyInt(c, constant.ContextKeyStickySessionTTL)
                        sessionManager.UpdateLastUsed(userId, modelRequest.Model, usingGroup, channelId, time.Duration(ttl)*time.Second)
                    } else {
                        // 渠道不可用，解绑并重新选择
                        sessionManager.UnbindChannel(userId, modelRequest.Model, usingGroup)
                        channel = nil
                    }
                }

                // 如果没有绑定或绑定的渠道不可用，选择新渠道
                if channel == nil {
                    channel, selectGroup, err = service.CacheGetRandomSatisfiedChannel(c, usingGroup, modelRequest.Model, 0)
                    if err == nil && channel != nil {
                        // 绑定新渠道到会话
                        ttl := common.GetContextKeyInt(c, constant.ContextKeyStickySessionTTL)
                        sessionManager.BindChannel(userId, modelRequest.Model, usingGroup, channel.Id, time.Duration(ttl)*time.Second)
                        common.SetContextKey(c, constant.ContextKeyStickySessionNew, true)
                    }
                }
            } else {
                // 原有逻辑：随机选择渠道
                channel, selectGroup, err = service.CacheGetRandomSatisfiedChannel(c, usingGroup, modelRequest.Model, 0)
            }

            // ... 错误处理
        }

        // ... 后续代码
    }
}

// 从请求中提取会话标识
func getSessionUserId(c *gin.Context, modelRequest *ModelRequest) string {
    // 优先使用请求中的 user 字段
    var request dto.GeneralOpenAIRequest
    if err := common.UnmarshalBodyReusable(c, &request); err == nil && request.User != "" {
        return request.User
    }

    // 降级方案：使用 Token ID
    tokenId := common.GetContextKeyInt(c, constant.ContextKeyTokenId)
```

### 性能影响评估

**Redis 性能**：
- 每次请求增加 1 次 Redis GET 操作
- 新会话增加 1 次 Redis SET 操作
- 预计延迟增加：<5ms

**内存占用**（内存缓存模式）：
- 每个会话约 100 字节
- 10万活跃会话 ≈ 10MB 内存

**结论**：✅ 性能影响可忽略不计

## 🛡️ 功能 2: 客户端限制（Client Restriction）

### 现状分析

**当前 New API 没有客户端识别功能**

New API 当前：
- ✅ 支持 IP 白名单 (Token.AllowIps)
- ❌ 不支持 User-Agent 限制
- ❌ 不支持客户端类型识别

### 实现方案

#### 核心设计

**1. 扩展 Token 模型**

```go
// model/token.go
type Token struct {
    // ... 现有字段

    // 客户端限制配置
    ClientRestrictionEnabled bool   `json:"client_restriction_enabled" gorm:"default:0"`
    AllowedClients          *string `json:"allowed_clients" gorm:"type:text;default:''"`
    // 格式：每行一个规则，支持通配符
    // 示例：
    // Claude-Code-CLI/*
    // VSCode/*
    // curl/7.*
}

// GetAllowedClientsMap 解析允许的客户端列表
func (token *Token) GetAllowedClientsMap() []string {
    if token.AllowedClients == nil || *token.AllowedClients == "" {
        return []string{}
    }

    cleanClients := strings.ReplaceAll(*token.AllowedClients, " ", "")
    clients := strings.Split(cleanClients, "\n")

    var result []string
    for _, client := range clients {
        client = strings.TrimSpace(client)
        if client != "" && !strings.HasPrefix(client, "#") {
            result = append(result, client)
        }
    }
    return result
}

// IsClientAllowed 检查客户端是否被允许
func (token *Token) IsClientAllowed(userAgent string) bool {
    if !token.ClientRestrictionEnabled {
        return true // 未启用限制，允许所有客户端
    }

    allowedClients := token.GetAllowedClientsMap()
    if len(allowedClients) == 0 {
        return true // 没有配置规则，允许所有客户端
    }

    for _, pattern := range allowedClients {
        if matchUserAgent(userAgent, pattern) {
            return true
        }
    }

    return false
}

// matchUserAgent 匹配 User-Agent 规则（支持通配符）
func matchUserAgent(userAgent, pattern string) bool {
    // 精确匹配
    if userAgent == pattern {
        return true
    }

    // 通配符匹配
    // pattern: "Claude-Code-CLI/*" 匹配 "Claude-Code-CLI/1.0.0"
    if strings.HasSuffix(pattern, "/*") {
        prefix := strings.TrimSuffix(pattern, "/*")
        return strings.HasPrefix(userAgent, prefix+"/")
    }

    // 正则表达式匹配（可选）
    if strings.HasPrefix(pattern, "regex:") {
        regexPattern := strings.TrimPrefix(pattern, "regex:")
        if matched, err := regexp.MatchString(regexPattern, userAgent); err == nil && matched {
            return true
        }
    }

    return false
}
```

**2. 在认证中间件中验证客户端**

```go
// middleware/auth.go
func TokenAuth() func(c *gin.Context) {

                    "type":    "invalid_request_error",

        }

        c.Next()
    }
}
```

### 客户端识别策略

**常见客户端 User-Agent 特征**：

```go
// service/client_detector.go
package service

type ClientType string

const (
    ClientTypeOfficialCLI     ClientType = "official_cli"      // Claude Code CLI
    ClientTypeVSCode          ClientType = "vscode"            // VS Code
    ClientTypeCursor          ClientType = "cursor"            // Cursor IDE
    ClientTypeWindsurf        ClientType = "windsurf"          // Windsurf
    ClientTypeCurl            ClientType = "curl"              // curl
    ClientTypePython          ClientType = "python"            // Python requests
    ClientTypePostman         ClientType = "postman"           // Postman
    ClientTypeUnknown         ClientType = "unknown"
)

var clientPatterns = map[ClientType][]string{
    ClientTypeOfficialCLI: {
        "Claude-Code-CLI/*",
        "claude-code/*",
    },
    ClientTypeVSCode: {
        "VSCode/*",
        "Visual Studio Code/*",
    },
    ClientTypeCursor: {
        "Cursor/*",
    },
    ClientTypeWindsurf: {
        "Windsurf/*",
    },
    ClientTypeCurl: {
        "curl/*",
    },
    ClientTypePython: {
        "python-requests/*",
        "Python/*",
    },
    ClientTypePostman: {
        "PostmanRuntime/*",
    },
}

func DetectClientType(userAgent string) ClientType {
    for clientType, patterns := range clientPatterns {
        for _, pattern := range patterns {
            if matchUserAgent(userAgent, pattern) {
                return clientType
            }
        }
    }
    return ClientTypeUnknown
}
```

## 🔍 潜在问题与解决方案

### 粘性会话

**问题 1：Redis 单点故障**
- **影响**：Redis 不可用时无法使用粘性会话
- **解决方案**：实现内存缓存降级，自动切换到本地缓存
- **代码**：已在 SessionManager 中实现

**问题 2：分布式环境下的会话同步**
- **影响**：多实例部署时需要共享会话状态
- **解决方案**：强制使用 Redis，不使用内存缓存
- **配置**：环境变量检查，部署文档说明

**问题 3：会话 TTL 过长导致渠道利用不均**
- **影响**：某些渠道长期绑定，其他渠道空闲
- **解决方案**：
  - 合理设置默认 TTL（建议 1 小时）
  - 提供手动解绑 API
  - 定期清理过期会话

**问题 4：用户不传 user 字段**
- **影响**：无法识别会话
- **解决方案**：降级到 Token ID 作为会话标识

### 客户端限制

**问题 1：User-Agent 可以伪造**
- **影响**：恶意用户可以伪造 UA 绕过限制
- **解决方案**：
  - 这是行业通用限制，无法完全防止
  - 作为第一层防御，配合其他安全措施（IP 限制、限流）
  - 对于高安全需求场景，使用更强的认证机制

**问题 2：客户端更新导致 UA 变化**
- **影响**：通配符规则可能失效
- **解决方案**：
  - 使用灵活的通配符（如 `Claude-Code-CLI/*`）
  - 定期检查和更新规则
  - 提供规则测试工具

**问题 3：正则表达式性能**
- **影响**：复杂正则可能影响性能
- **解决方案**：
  - 缓存编译后的正则表达式
  - 限制正则复杂度
  - 优先使用简单通配符匹配
